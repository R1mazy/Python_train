a = int(input())
b = int(input())
min_n = min(a, b)
max_n = max(a, b)
stop = True

#НОД
for i in range(1, min_n + 1):
    if ((a % i == 0) and (b % i == 0)):
        nod = i

print("НОД введенных чисел: " + str(nod))

#НОК
while stop == True:
    if ((max_n % a == 0) and (max_n % b == 0)):
        nok = max_n
        stop = False
    max_n += 1

print("НОК введенных чисел: " + str(nok))
